namespace = suppression_event
suppression_event.1 = {
	type = country_event
	hidden = yes
	trigger = {
		OR = {
			suppression_manufacturing_trigger = yes
			suppression_agricultural_trigger = yes
			suppression_resource_trigger = yes
		}
	}
	immediate = {
		if = {
			limit = {
				suppression_manufacturing_trigger = yes
			}
			every_scope_building = {
				limit = {
					AND = {
						suppression_manufacturing_bg_trigger = yes
						not = {
							has_modifier = suppression_modifier
						}
					}
				}
				add_modifier = suppression_modifier
			}
		}
		if = {
			limit = {
				suppression_agricultural_trigger = yes
			}
			every_scope_building = {
				limit = {
					AND = {
						suppression_agricultural_bg_trigger = yes
						not = {
							has_modifier = suppression_modifier
						}
					}
				}
				add_modifier = suppression_modifier
			}
		}
		if = {
			limit = {
				suppression_resource_trigger = yes
			}
			every_scope_building = {
				limit = {
					AND = {
						suppression_resource_bg_trigger = yes
						not = {
							has_modifier = suppression_modifier
						}
					}
				}
				add_modifier = suppression_modifier
			}
		}
	}
}

suppression_event.2 = {
	type = country_event
	hidden = yes
	trigger = {
		OR = {
			suppression_manufacturing_trigger = no
			suppression_agricultural_trigger = no
			suppression_resource_trigger = no
		}
	}
	immediate = {
		if = {
			limit = {
				suppression_manufacturing_trigger = no
			}
			every_scope_building = {
				limit = {
					AND = {
						suppression_manufacturing_bg_trigger = yes
						has_modifier = suppression_modifier
					}
				}
				remove_modifier = suppression_modifier
			}
		}
		if = {
			limit = {
				suppression_agricultural_trigger = no
			}
			every_scope_building = {
				limit = {
					AND = {
						suppression_agricultural_bg_trigger = yes
						has_modifier = suppression_modifier
					}
				}
				remove_modifier = suppression_modifier
			}
		}
		if = {
			limit = {
				suppression_resource_trigger = no
			}
			every_scope_building = {
				limit = {
					AND = {
						suppression_resource_bg_trigger = yes
						has_modifier = suppression_modifier
					}
				}
				remove_modifier = suppression_modifier
			}
		}
	}
}
