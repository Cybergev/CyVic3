namespace = suppression_event
suppression_event.1 = {
	type = country_event
	hidden = yes
	trigger = {
		OR = {
			has_modifier = suppression_colonial_manufacturing_industry_policy
			has_modifier = suppression_colonial_agricultural_industry_policy
			has_modifier = suppression_colonial_resource_industry_policy
		}
	}
	immediate = {
		every_country = {
			every_scope_building = {
				if = {
					limit = {
						not = {
							has_modifier = suppression_modifier
						}
						OR = {
							state = {
								owner = {
									this = root
								}
								is_incorporated = no
							}
							owner = {
								OR = {
									has_diplomatic_pact = {
										who = root
										type = puppet
									}
									has_diplomatic_pact = {
										who = root
										type = colony
									}
									has_diplomatic_pact = {
										who = root
										type = vassal
									}
								}
							}
						}
					}
					if = {
						limit = {
							owner = {
								root = {
									has_modifier = suppression_colonial_manufacturing_industry_policy
								}
							}
							suppression_manufacturing_bg_trigger = yes
						}
						add_modifier = suppression_modifier
					}
					if = {
						limit = {
							owner = {
								root = {
									has_modifier = suppression_colonial_agricultural_industry_policy
								}
							}
							suppression_agricultural_bg_trigger = yes
						}
						add_modifier = suppression_modifier
					}
					if = {
						limit = {
							owner = {
								root = {
									has_modifier = suppression_colonial_resource_industry_policy
								}
							}
							suppression_resource_bg_trigger = yes
						}
						add_modifier = suppression_modifier
					}
				}
			}
		}
	}
}
