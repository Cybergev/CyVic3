REPLACE_OR_CREATE:suppression_trigger = {
	OR = {
		has_modifier = suppression_colonial_manufacturing_industry_policy
		AND = {
			OR = {
				is_subject_type = subject_type_puppet
				is_subject_type = subject_type_colony
				is_subject_type = subject_type_vassal
				is_subject_type = subject_type_crown_land
			}
			any_overlord_or_above = {
				has_modifier = suppression_colonial_manufacturing_industry_policy
			}
		}
	}
}

REPLACE_OR_CREATE:suppression_manufacturing_trigger = {
	not = {
		has_modifier = suppression_modifier
	}
	OR = {
		state = {
			is_incorporated = yes
			owner = {
				any_overlord_or_above = {
					has_modifier = suppression_colonial_manufacturing_industry_policy
				}
			}
		}
		state = {
			is_incorporated = no
			owner = {
				OR = {
					has_modifier = suppression_colonial_manufacturing_industry_policy
					any_overlord_or_above = {
						has_modifier = suppression_colonial_manufacturing_industry_policy
					}
				}
			}
		}
	}
}
