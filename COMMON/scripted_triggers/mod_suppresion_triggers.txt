REPLACE_OR_CREATE:suppression_manufacturing_bg_trigger = {
	OR = {
		is_building_group = bg_manufacturing
		is_building_type = building_arts_academy
	}
}

REPLACE_OR_CREATE:suppression_manufacturing_trigger = {
	OR = {
		AND = {
			owner = {
				has_modifier = suppression_colonial_manufacturing_industry_policy
			}
			is_incorporated = no
		}
		owner = {
			any_overlord_or_above = {
				has_modifier = suppression_colonial_manufacturing_industry_policy
				OR = {
					has_diplomatic_pact = {
						who = PREV
						type = puppet
					}
					has_diplomatic_pact = {
						who = PREV
						type = colony
					}
					has_diplomatic_pact = {
						who = PREV
						type = vassal
					}
				}
			}
		}
	}
}

REPLACE_OR_CREATE:suppression_agricultural_bg_trigger = {
	OR = {
		is_building_group = bg_subsistence_agriculture
		is_building_group = bg_subsistence_ranching
		is_building_group = bg_agriculture
		is_building_group = bg_plantations
		is_building_group = bg_ranching
	}
}

REPLACE_OR_CREATE:suppression_agricultural_trigger = {
	any_scope_state = {
		OR = {
			owner = {
				any_overlord_or_above = {
					AND = {
						has_modifier = suppression_colonial_agricultural_industry_policy
						OR = {
							has_diplomatic_pact = {
								who = PREV
								type = puppet
							}
							has_diplomatic_pact = {
								who = PREV
								type = colony
							}
							has_diplomatic_pact = {
								who = PREV
								type = vassal
							}
						}
					}
				}
			}
			AND = {
				owner = {
					has_modifier = suppression_colonial_agricultural_industry_policy
				}
				is_incorporated = no
			}
		}
		any_scope_building = {
			suppression_agricultural_bg_trigger = yes
			not = {
				has_modifier = suppression_modifier
			}
		}
	}
}
