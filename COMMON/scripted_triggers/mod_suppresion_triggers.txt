REPLACE_OR_CREATE:suppression_manufacturing_trigger = {
	trigger_if = {
		limit = {
			suppression_manufacturing_bg_trigger = yes
		}
		NOT = {
			has_modifier = suppression_modifier
		}
		OR = {
			state = {
				is_incorporated = yes
				owner = {
					suppressible_subject_trigger = yes
					any_overlord_or_above = {
						has_modifier = suppression_colonial_manufacturing_industry_policy
					}
				}
			}
			state = {
				is_incorporated = no
				owner = {
					OR = {
						has_modifier = suppression_colonial_manufacturing_industry_policy
						AND = {
							suppressible_subject_trigger = yes
							any_overlord_or_above = {
								has_modifier = suppression_colonial_manufacturing_industry_policy
							}
						}
					}
				}
			}
		}
	}
	trigger_else = {
		always = no
	}
}
