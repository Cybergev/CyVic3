promote_national_val = {
	add_modifier = {
		name = national_values_promotion
		years = 5
		is_decaying = yes
	}
	random_scope_state = {
		limit = {
			is_capital = yes
		}
		random_scope_pop = {
			limit = {
				culture = {
					is_primary_culture_of = ROOT
				}
				religion = {
					is_state_religion = ROOT
				}
			}
			culture = {
				save_scope_as = state_cul
			}
			religion = {
				save_scope_as = state_rel
			}
		}
	}
	every_scope_pop = {
		limit = {
			NOT = {
				culture = scope:state_cul
			}
		}
		change_pop_culture = {
			target = scope:state_cul
			value = 0.01
		}
		change_pop_religion = {
			target = scope:state_rel
			value = 0.01
		}
	}
}
