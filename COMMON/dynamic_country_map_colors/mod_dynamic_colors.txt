use_monarchy = {
	color = rgb { 50 150 150 }
	possible = {
		exists = c:USE
		THIS = c:USE
		has_law = law_type:law_monarchy
	}
}

use_republic = {
	color = { 10 50 100 }
	possible = {
		exists = c:USE
		THIS = c:USE
		OR = {
			has_law = law_type:law_presidential_republic
			has_law = law_type:law_parliamentary_republic
		}
	}
}

use_technate = {
	color = rgb { 80 80 80 }
	possible = {
		exists = c:USE
		THIS = c:USE
		has_law = law_type:law_technocracy
	}
}

use_communist = {
	color = rgb { 100 10 50 }
	possible = {
		exists = c:USE
		THIS = c:USE
		has_law = law_type:law_council_republic
	}
}

use_fascist = {
	color = rgb { 50 50 50 }
	possible = {
		exists = c:USE
		THIS = c:USE
		any_scope_character = {
			is_ruler = yes
			OR = {
				has_ideology = ideology:ideology_fascist
				has_ideology = ideology:ideology_integralist
			}
		}
	}
}
