movement_positivist = {
	category = movement_category_ideological
	ideology = ideology_positivism
	character_ideologies = {
		ideology_positivist
	}
	creation_trigger = {
		has_variable = positivism_invented
		has_technology_researched = empiricism
	}
	creation_weight = {
		value = 10
		if = {
			limit = {
				has_technology_researched = philosophical_pragmatism
			}
			add = {
				value = 100
			}
		}
	}
	on_created = {
		add_movement_enthusiasm_modifier = yes
	}
	character_support_trigger = {
		trigger_if = {
			limit = {
				is_in_exile_pool = yes
			}
			OR = {
				has_ideology = ideology:ideology_positivist
				has_ideology = ideology:ideology_atheist
				has_ideology = ideology:ideology_republican_leader
				is_interest_group_type = ig_armed_forces
			}
		}
		trigger_else = {
			OR = {
				has_ideology = ideology:ideology_positivist
				has_ideology = ideology:ideology_atheist
				has_ideology = ideology:ideology_republican_leader
				interest_group ?= {
					OR = {
						is_interest_group_type = ig_armed_forces
						has_ideology = ideology:ideology_oligarchic
					}
				}
			}
		}
	}
	character_support_weight = {
		value = 200
		if = {
			limit = {
				OR = {
					has_ideology = ideology:ideology_positivist
				}
			}
			multiply = {
				value = 5
			}
		}
	}
	pop_support_trigger = {
		NOR = {
			is_pop_type = aristocrats
			is_pop_type = clergymen
		}
	}
	pop_support_factors = {
		movement_support_high_urbanization
		movement_support_high_literacy
		movement_support_capitalists
		movement_support_academics
		movement_support_bureaucrats
		movement_support_engineers
		movement_support_officers
		movement_support_clerks
		movement_support_machinists
		movement_support_soldiers
		movement_support_je_positivist_movement
	}
	pop_support_weight = {
		add = {
			desc = "URBAN_STATE"
			value = 10
			multiply = state.state_urbanization_rate
		}
		if = {
			limit = {
				strata = upper
			}
			if = {
				limit = {
					is_pop_type = capitalists
				}
				add = {
					value = 5
					desc = "POP_CAPITALISTS"
				}
			}
		}
		else_if = {
			limit = {
				strata = middle
			}
			else_if = {
				limit = {
					is_pop_type = academics
				}
				add = {
					value = 20
					desc = "POP_ACADEMICS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = bureaucrats
				}
				add = {
					value = 20
					desc = "POP_BUREAUCRATS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = engineers
				}
				add = {
					value = 20
					desc = "POP_ENGINEERS"
				}
			}
			if = {
				limit = {
					is_pop_type = officers
				}
				if = {
					limit = {
						owner ?= {
							OR = {
								has_law = law_type:law_professional_army
								has_law = law_type:law_mass_conscription
							}
						}
					}
					add = {
						value = 40
						desc = "POP_OFFICERS"
					}
				}
				else = {
					add = {
						value = 20
						desc = "POP_OFFICERS"
					}
				}
			}
			else = {
				add = {
					value = 5
					desc = "MIDDLE_NO_ICON"
				}
			}
		}
		else_if = {
			limit = {
				strata = lower
			}
			if = {
				limit = {
					is_pop_type = clerks
				}
				add = {
					value = 10
					desc = "POP_CLERKS"
				}
			}
			if = {
				limit = {
					is_pop_type = machinists
				}
				add = {
					value = 10
					desc = "POP_MACHINISTS"
				}
			}
			else_if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 10
					desc = "POP_SOLDIERS"
				}
			}
			else = {
				add = {
					value = 5
					desc = "LOWER_NO_ICON"
				}
			}
		}
		if = {
			limit = {
				owner ?= {
					has_variable = positivist_leaders_increase_var
				}
			}
			if = {
				limit = {
					is_pop_type = officers
				}
				multiply = {
					value = 1.25
					desc = "POSITIVISM_BONUS"
				}
			}
			if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 1.1
					desc = "POSITIVISM_BONUS"
				}
			}
		}
		if = {
			limit = {
				owner ?= {
					has_variable = positivist_leaders_large_increase_var
				}
			}
			if = {
				limit = {
					is_pop_type = officers
				}
				multiply = {
					value = 1.5
					desc = "POSITIVISM_BONUS"
				}
			}
			if = {
				limit = {
					is_pop_type = soldiers
				}
				add = {
					value = 1.25
					desc = "POSITIVISM_BONUS"
				}
			}
		}
		multiply = {
			# Illiterate pops participate at 1/10 the rate of fully literate pops
			desc = "POP_LITERACY"
			value = literacy_rate
			add = 0.1
		}
	}
	revolution = {
		possible = {
			political_movement_support >= 0.05
		}
		weight = {
			value = 100
		}
		state_weight = {
			value = 1
		}
	}
	secession = {
		possible = {
			political_movement_support >= 0.05
			scope:clout <= 0
		}
		weight = {
			value = 1000
		}
	}
}
