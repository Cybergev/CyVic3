#ideolog swaps
swap_ideology_to_anarchist = {
	icon = "gfx/interface/icons/ideology_icons/ideology_leader/ideology_leader_anarchist.dds"
	clicksound = "event:/sfx/dlc/1.3_ip1/ui/character_invite"
	potential = {
		owner = {
			scope:actor = owner
		}
		or = {
			has_role = ruler
			has_role = politician
		}
		interest_group = {
			is_in_government = yes
		}
	}
	show_requirements = yes
	possible = {
		or = {
			has_ideology = ideology:ideology_communist
			and = {
				has_ideology = ideology:ideology_moderate
				or = {
					owner = {
						has_law = law_type:law_council_republic
					}
					interest_group = {
						or = {
							has_ideology = ideology:ideology_anarchist_movement
							has_ideology = ideology:ideology_communist
						}
					}
				}
			}
		}
	}
	effect = {
		set_ideology = ideology:ideology_anarchist
		add_modifier = {
			name = changed_ideology
			years = 5
			is_decaying = yes
		}
	}
	ai_chance = {
		base = 0
	}
}
