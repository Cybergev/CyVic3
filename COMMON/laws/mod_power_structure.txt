law_chiefdom = {
	group = lawgroup_governance_principles
	icon = "gfx/interface/icons/law_icons/chiefdom.dds"
	progressiveness = -200
	can_enact = {
		is_country_type = decentralized
	}
	is_visible = {
		is_country_type = decentralized
	}
	modifier = {
		country_authority_add = 200
		country_legitimacy_headofstate_add = 50
		country_legitimacy_govt_size_add = 2
	}
	on_activate = {
		if = {
			limit = {
				NOT = {
					is_country_type = decentralized
				}
			}
			activate_law = law_type:law_monarchy
		}
	}
	ai_enact_weight_modifier = {
		#Petitions
		value = 0
		if = {
			limit = {
				ai_has_enact_weight_modifier_journal_entries = yes
			}
			add = 750
		}
	}
}

law_theocracy = {
	group = lawgroup_governance_principles
	icon = "gfx/interface/icons/law_icons/theocracy.dds"
	progressiveness = -100
	disallowing_laws = {
		law_total_separation
	}
	modifier = {
		country_authority_add = 200
		country_legitimacy_headofstate_add = 50
		country_legitimacy_ideological_incoherence_mult = 0.1
		state_conversion_mult = 0.25
		country_clergymen_pol_str_mult = 0.25
	}
	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology:ideology_theocrat
		}
	}
	ai_enact_weight_modifier = {
		#Petitions
		value = 0
		if = {
			limit = {
				ai_has_enact_weight_modifier_journal_entries = yes
			}
			add = 750
		}
	}
	ai_impose_chance = {
		value = 0
		if = {
			limit = {
				has_law = law_type:law_theocracy
				has_strategy = ai_strategy_reactionary_agenda
				power_bloc ?= {
					has_identity = identity:identity_religious
				}
				religion = scope:target_country.religion
				scope:target_country = {
					# Is their law different enough from what we want to impose?					
					NOT = {
						has_law = law_type:law_monarchy
					}
				}
			}
			add = base_impose_law_weight
			if = {
				limit = {
					scope:target_country = {
						has_law = law_type:council_republic
					}
					NOT = {
						has_strategy = ai_strategy_egalitarian_agenda
					}
				}
				multiply = 5
			}
			else = {
				multiply = 0.5				# AI is less likely to mess with governance principles unless dealing with council republics
			}
		}
		if = {
			limit = {
				scope:target_country = {
					c:BIC ?= this
					OR = {
						has_journal_entry = je_british_dictate_plantations
						has_journal_entry = je_british_dictate_military
						has_journal_entry = je_british_dictate_universities
					}
				}
			}
			multiply = 0
		}
	}
}

	}
	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country

					modifier:country_can_impose_same_lawgroup_governance_principles_in_power_bloc_bool = yes

				c:FRA ?= this

		law_parliamentary_republic

		country_legitimacy_ideological_incoherence_mult = 0.5

				text = BIC_not_able_to_change_distribution_of_power_tt

			}
		}
	}
	can_impose = {
		OR = {
			AND = {
				is_in_same_power_bloc = scope:target_country
				AND = {
					modifier:country_can_impose_same_lawgroup_distribution_of_power_in_power_bloc_bool = yes
					has_law = scope:law
				}
			}
			can_impose_law_default = yes
		}
	}
	ai_enact_weight_modifier = {
		#Petitions

			limit = {
				ai_has_enact_weight_modifier_journal_entries = yes
