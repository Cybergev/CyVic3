law_bakufu = {
	group = lawgroup_distribution_of_power
	parent = law_autocracy
	icon = "gfx/interface/icons/law_icons/bakufu.dds"
	progressiveness = -100
	unlocking_laws = {
		law_monarchy
		law_social_monarchy
	}
	is_visible = {
		c:JAP ?= this
		NOT = {
			has_law = law_type:law_autocracy
		}
	}
	on_activate = {
		scripted_effect_parties_disappearence = yes
		if = {
			limit = {
				country_has_monarchy_law = yes
				game_date > 1836.1.1
				NOR = {
					# Revolter tags are handled a different way
					is_revolutionary = yes
					is_secessionist = yes
				}
			}
			custom_tooltip = {
				text = shogun_returns_to_throne_tt
				country_generate_japanese_shogun = yes
			}
		}
		hidden_effect = {
			country_assess_daimyo_alignments = yes
		}
	}
	on_deactivate = {
		if = {
			limit = {
				country_has_monarchy_law = yes
				country_has_non_monarchy_law = no				# To handle an execution order issue
			}
			custom_tooltip = {
				text = emperor_returns_to_throne_tt
				country_restore_japanese_emperor = yes
			}
			hidden_effect = {
				country_assess_daimyo_alignments = yes
			}
		}
	}
	modifier = {
		country_authority_add = 200
		country_legitimacy_govt_total_clout_add = 100
		country_legitimacy_headofstate_add = 50
		country_legitimacy_ideological_incoherence_mult = 0.5
		country_legitimacy_govt_size_add = 1
		country_aristocrats_pol_str_mult = 0.5
		country_officers_pol_str_mult = 1
		country_soldiers_pol_str_mult = 0.5
	}
	can_impose = {
		always = no
	}
	ai_enact_weight_modifier = {
		#Petitions
		value = 0
	}
	ai_impose_chance = {
		value = 0
	}
}

law_sakoku = {
	group = lawgroup_trade_policy
	parent = law_isolationism	# Inherits ideological stances from parent
	icon = "gfx/interface/icons/law_icons/sakoku.dds"
	progressiveness = -50
	requires_law_or = {
		law_traditionalism
	}
	is_visible = {
		country_has_primary_culture = cu:japanese
		any_scope_state = {
			state_region = s:STATE_KYUSHU
			is_largest_state_in_region = yes
		}
	}
	modifier = {
		country_authority_mult = 0.25
		country_influence_mult = -0.5
		country_tech_spread_mult = -0.2
		state_trade_capacity_mult = -0.75
		state_tariff_import_add = 1
		state_tariff_export_add = 1
		state_subvention_export_add = 0
		state_subvention_import_add = 0
		country_leverage_resistance_mult = 0.75
		country_disallow_trade_outside_kyushu_bool = yes
		state_trade_center_max_limit_add = 5
		country_disallow_aggressive_plays_bool = yes
	}
	ai_will_do = {
		OR = {
			has_journal_entry = je_sakoku
			ruler = {
				OR = {
					has_ideology = ideology:ideology_traditionalist
					has_ideology = ideology:ideology_vanguardist
				}
			}
		}
	}
	on_deactivate = {
		if = {
			limit = {
				has_law = law_type:law_closed_borders
			}
			custom_tooltip = internal_controls_relaxed_tt
			activate_law = law_type:law_migration_controls
		}
		custom_tooltip = {
			text = landowners_stop_supporting_sakoku_tt
			ig:ig_landowners ?= {
				remove_ideology = ideology_japan_hierarchic
				add_ideology = ideology_japan_hierarchic_2
			}
		}
	}
	ai_enact_weight_modifier = {
		#Petitions
		value = 0
		if = {
			limit = {
				ai_has_enact_weight_modifier_journal_entries = yes
			}
			add = 750
		}
	}
}
