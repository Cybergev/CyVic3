	is_shown = {

		}
	}
	ai_chance = {
		base = 0
	}
}

homeland_expansion = {
	is_shown = {
		always = yes
	}
	possible = {
		has_technology_researched = pan-nationalism
		not = {
			has_modifier = homeland_expanded
		}
		any_scope_state = {
			is_incorporated = yes
			is_split_state = no
			owner = root
			has_decree = decree_promote_national_values
			any_scope_culture = {
				has_homeland = prev
				not = {
					culture_is_discriminated_in = prev.owner
					is_primary_culture_of = prev.owner
				}
			}
		}
	}
	when_taken = {
		every_primary_culture = {
			save_scope_as = adition_culture
		}
		random_scope_state = {
			limit = {
				is_incorporated = yes
				is_split_state = no
				owner = root
				has_decree = decree_promote_national_values
				any_scope_culture = {
					has_homeland = prev
					not = {
						culture_is_discriminated_in = prev.owner
						is_primary_culture_of = prev.owner
					}
				}
			}
			every_scope_culture = {
				limit = {
					has_homeland = prev
					not = {
						culture_is_discriminated_in = prev.owner
						is_primary_culture_of = prev.owner
					}
				}
				prev.state_region = {
					remove_homeland = prev
					add_radicals_in_state = {
						value = 1
						culture = prev
					}
				}
			}
			scope:adition_culture = {
				prev.state_region = {
					add_homeland = prev
					add_loyalists_in_state = {
						value = 0.5
						culture = prev
					}
				}
			}
		}
		add_modifier = {
			name = homeland_expanded
			years = 5
			is_decaying = yes
		}
	}
	ai_chance = {
		base = 0
	}
}

	is_shown = {

			limit = {

					country_has_state_religion = rel:atheist
				}
			}
			set_state_religion = rel:atheist

		if = {
			limit = {
				not = {
					has_law = law_type:law_state_atheism

				country_has_state_religion = rel:atheist
			}
			set_state_religion = root.country_definition.religion
		}
		promote_national_val = yes
	}
	ai_chance = {
		base = 1000
	}
}
