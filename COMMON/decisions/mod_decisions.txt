	is_shown = {
		c:USE ?= root

	}
	possible = {
		c:USE ?= root
		NOT = {
			has_variable = european_homeland_unificated
		}

			OR = {
				region = sr:region_north_sea_coast
				region = sr:region_england
				region = sr:region_iberia
				region = sr:region_occitania
				region = sr:region_france
				region = sr:region_rhine
				region = sr:region_baltic

				region = sr:region_italy	
				region = sr:region_balkans
				region = sr:region_danubia
				region = sr:region_poland
				region = sr:region_dnieper
				region = sr:region_russia
				region = sr:region_caucasus
			}
			is_incorporated = yes
			is_split_state = no
			owner = root
			has_decree = decree_promote_national_values
			cu:european = {
				not = {
					has_homeland = prev
				}
			}
		}
	}
	when_taken = {
		random_scope_state = {
			limit = {
				OR = {
					region = sr:region_north_sea_coast
					region = sr:region_england
					region = sr:region_iberia
					region = sr:region_occitania
					region = sr:region_france
					region = sr:region_rhine
					region = sr:region_baltic	
					region = sr:region_north_germany
					region = sr:region_south_germany
					region = sr:region_italy	
					region = sr:region_balkans
					region = sr:region_danubia
					region = sr:region_poland
					region = sr:region_dnieper
					region = sr:region_russia
					region = sr:region_caucasus
				}
				is_incorporated = yes
				is_split_state = no
				owner = root
				has_decree = decree_promote_national_values
				cu:european = {
					not = {
						has_homeland = prev
					}
				}
			}
			every_scope_culture = {
				limit = {
					has_homeland = prev
				}
				prev.state_region = {
					remove_homeland = prev
					add_homeland = cu:european
					add_radicals = {
						value = 1
						culture = prev
					}
				}
			}
		}
		set_variable = {
			name = european_homeland_unificated
			value = 1
			days = 180
		}
	}
	ai_chance = {
		base = 0
	}
}

socialistic_reform = {
	is_shown = {
		has_law = law_type:law_council_republic
	}
	possible = {
		ruler.interest_group.party = {
			is_party_type = communist_party
		}
		any_interest_group = {
			is_in_government = yes
			OR = {
				not = {
					is_interest_group_type = ig_landowners
				}
				not = {
					is_interest_group_type = ig_industrialists
				}
				not = {
					is_interest_group_type = ig_devout
				}
			}
			OR = {
				not = {
					has_ideology = ideology:ideology_proletarian
				}
				not = {
					has_ideology = ideology:ideology_socialist
				}
			}
		}
	}
	when_taken = {
		ROOT = {
			every_interest_group = {
				if = {
					limit = {
						is_in_government = yes
						OR = {
							not = {
								is_interest_group_type = ig_landowners
							}
							not = {
								is_interest_group_type = ig_industrialists
							}
							not = {
								is_interest_group_type = ig_devout
							}
						}
						OR = {
							not = {
								has_ideology = ideology:ideology_proletarian
							}
							not = {
								has_ideology = ideology:ideology_socialist
							}
						}
					}
					if = {
						limit = {
							has_ideology = ideology:ideology_agrarian
						}
						remove_ideology = ideology_agrarian
						add_ideology = ideology_proletarian
						add_ideology = ideology_socialist
					}
					if = {
						limit = {
							has_ideology = ideology:ideology_republican
						}
						remove_ideology = ideology_republican
						add_ideology = ideology_proletarian
						add_ideology = ideology_socialist
					}
					if = {
						limit = {
							has_ideology = ideology:ideology_reactionary
						}
						remove_ideology = ideology_reactionary
						add_ideology = ideology_proletarian
						add_ideology = ideology_socialist
					}
				}
			}
			hidden_effect = {
				every_interest_group = {
					if = {
						limit = {
							or = {
								has_ideology = ideology:ideology_proletarian
								has_ideology = ideology:ideology_socialist
							}
						}
						remove_ideology = ideology_proletarian
						remove_ideology = ideology_socialist
						add_ideology = ideology_proletarian
						add_ideology = ideology_socialist
					}
				}
			}
		}
	}
	ai_chance = {
		base = 0
	}
}

				days = 180
			}

			limit = {

					OR = {
						has_law = law_type:law_state_atheism

							country_has_state_religion = rel:atheist
						}
					}
					OR = {
						NOT = {

						}
						country_has_state_religion = rel:atheist
					}
				}
			}
			activate_law = law_type:law_state_atheism
		}
	}
	ai_chance = {

	}
}
