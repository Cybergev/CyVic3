shinto_decision = {
	is_shown = {
		exists = c:JAP
		c:JAP = THIS
		NOT = {
			country_has_state_religion = rel:shinto
		}
	}
	possible = {
		custom_tooltip = {
			text = is_meiji_japan
			has_variable = meiji_restoration_complete
		}
		NOT = {
			has_law = law_type:law_state_atheism
		}
	}
	when_taken = {
		set_state_religion = rel:shinto
		state_religion_switch_effect = yes
		ig:ig_devout = {
			hidden_effect = {
				set_interest_group_name = ig_shinto_monks
			}
			remove_ideology = ideology_buddhist_moralist
			remove_ideology = ideology_pious
			add_ideology = ideology_shinto_moralist
			add_ideology = ideology_pious
			set_ig_trait = ig_trait:ig_trait_haibutsu_kishaku
			set_ig_trait = ig_trait:ig_trait_heavenly_sovereign
			set_ig_trait = ig_trait:ig_trait_secular_shrine_theory
			leader ?= {
				change_character_religion = rel:shinto
			}
		}
		random_scope_state = {
			limit = {
				owner = ROOT
				is_capital = yes
			}
			convert_population = {
				target = rel:shinto
				value = 0.4
			}
		}
		every_scope_state = {
			limit = {
				owner = ROOT
				is_incorporated = yes
				NOT = {
					is_capital = yes
				}
			}
			convert_population = {
				target = rel:shinto
				value = 0.2
			}
		}
	}
	ai_chance = {
		base = 100
	}
}
