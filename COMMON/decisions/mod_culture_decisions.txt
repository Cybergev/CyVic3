european_culture_unification = {
	is_shown = {
		european_culture_check = yes
		not = {
			country_has_primary_culture = cu:european
		}
	}
	possible = {
		not = {
			country_has_primary_culture = cu:european
		}
	}
	when_taken = {
		every_primary_culture = {
			root = {
				remove_primary_culture = prev
				add_radicals = {
					value = 1
					culture = prev
				}
			}
		}
		add_primary_culture = cu:european
	}
	ai_chance = {
		base = 0
	}
}

join_north_german_unification = {
	is_shown = {
		country_is_in_europe = yes
	}
	possible = {
		OR = {
			country_has_primary_culture = cu:danish
			country_has_primary_culture = cu:dutch
			country_has_primary_culture = cu:norwegian
			country_has_primary_culture = cu:swedish
		}
		NOT = {
			country_has_primary_culture = cu:north_german
		}
	}
	when_taken = {
		add_primary_culture = cu:north_german
	}
	ai_chance = {
		base = 0
	}
}
