mod_change_private_construction = {
	is_shown = {
		always = yes
	}
	when_taken = {
		if = {
			limit = {
				NOT = {
					has_modifier = ban_private_construction
					has_law = law_type:law_extraction_economy
					has_law = law_type:law_command_economy
				}
			}
			add_modifier = ban_private_construction
			add_radicals = {
				pop_type = aristocrats
				value = 1
			}
			add_radicals = {
				pop_type = capitalists
				value = 1
			}
		}
		else = {
			remove_modifier = ban_private_construction
		}
	}
	ai_chance = {
		base = 0
	}
}
