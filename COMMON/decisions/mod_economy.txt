mod_support_private_construction = {
	is_shown = {
		always = yes
	}
	possible = {
		NOT = {
			has_modifier = ban_private_construction
		}
	}
	when_taken = {
		if = {
			limit = {
				NOT = {
					has_modifier = support_private_construction
				}
			}
			add_modifier = support_private_construction
			add_loyalists = {
				pop_type = aristocrats
				value = 1
			}
			add_loyalists = {
				pop_type = capitalists
				value = 1
			}
		}
		else = {
			remove_modifier = support_private_construction
		}
	}
	ai_chance = {
		value = 0
	}
}
