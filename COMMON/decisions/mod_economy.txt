mod_support_private_construction = {
	is_shown = {
		always = yes
	}
	possible = {
		NOT = {
			has_modifier = ban_private_construction
		}
	}
	when_taken = {
		if = {
			limit = {
				NOT = {
					has_modifier = support_private_construction
				}
			}
			add_modifier = support_private_construction
			add_loyalists = {
				pop_type = aristocrats
				value = 1
			}
			add_loyalists = {
				pop_type = capitalists
				value = 1
			}
		}
		else = {
			remove_modifier = support_private_construction
		}
	}
	ai_chance = {
		value = 0
	}
}

suppress_colonial_manufacturing_industry = {
	is_shown = {
		not = {
			has_modifier = suppression_colonial_manufacturing_industry_policy
		}
	}
	possible = {
		not = {
			has_modifier = suppression_colonial_manufacturing_industry_policy
		}
	}
	when_taken = {
		add_modifier = suppression_colonial_manufacturing_industry_policy
		trigger_event = {
			id = suppression_event.1
		}
		every_country = {
			limit = {
				is_subject_of = root
			}
			trigger_event = {
				id = suppression_event.1
			}
		}
	}
	ai_chance = {
		value = 0
		modifier = {
			or = {
				has_strategy = ai_strategy_economic_imperialism
				has_strategy = ai_strategy_colonial_expansion
				has_strategy = ai_strategy_colonial_extraction
				has_strategy = ai_strategy_industrial_expansion
			}
			add = 100
		}
	}
}
