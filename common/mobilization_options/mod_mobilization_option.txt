mobilization_option_basic_supplies = {
	texture = "gfx/interface/icons/mobilization_options/basic_supplies.dds"
	possible = {
		scope:military_formation = {
			NOR = {
				has_mobilization_option = mobilization_option:mobilization_option_extra_supplies
				has_mobilization_option = mobilization_option:mobilization_option_luxurious_supplies
			}
		}
	}
	can_be_turned_off = {
		scope:military_formation = {
			OR = {
				has_mobilization_option = mobilization_option:mobilization_option_extra_supplies
				has_mobilization_option = mobilization_option:mobilization_option_luxurious_supplies
			}
		}
	}
	upkeep_modifier = {
		goods_input_grain_add = 1
	}
	unit_modifier = {
		# empty on purpose - this is the minimum mobilization cost
	}
	ai_weight = {
		value = 1
	}
	group = supplies
}

mobilization_option_extra_supplies = {
	texture = "gfx/interface/icons/mobilization_options/extra_supplies.dds"
	possible = {
		scope:military_formation = {
			NOT = {
				has_mobilization_option = mobilization_option:mobilization_option_luxurious_supplies
			}
		}
		market ?= {
			mg:fruit ?= {
				market_goods_sell_orders > 0
			}
			mg:groceries ?= {
				market_goods_sell_orders > 0
			}
		}
	}
	unlocking_technologies = {
		army_reserves
	}
	on_activate_while_mobilized = {
		custom_tooltip = {
			text = mobilization_option_it_hurts_organization_when_you_adjust_equipment_tt
			add_organization = {
				value = organization
				multiply = -0.5
			}
		}
	}
	on_deactivate = {
		custom_tooltip = {
			text = mobilization_option_it_hurts_morale_and_organization_when_you_remove_medic_support_tt
			every_combat_unit = {
				add_morale = {
					value = morale
					multiply = -0.5
				}
			}
			add_organization = {
				value = organization
				multiply = -0.25
			}
		}
	}
	on_deactivate_while_mobilized = {
		custom_tooltip = {
			text = mobilization_option_it_hurts_morale_and_organization_when_you_remove_medic_support_tt
			every_combat_unit = {
				add_morale = {
					value = morale
					multiply = -0.5
				}
			}
			add_organization = {
				value = organization
				multiply = -0.25
			}
		}
	}
	upkeep_modifier = {
		goods_input_fruit_add = 10
		goods_input_groceries_add = 10
	}
	upkeep_modifier_unscaled = {
		goods_input_small_arms_mult = 0.2
		goods_input_artillery_mult = 0.2
		goods_input_ammunition_mult = 0.2
		goods_input_oil_mult = 0.2
		goods_input_radios_mult = 0.2
		goods_input_tanks_mult = 0.2
		goods_input_automobiles_mult = 0.2
	}
	unit_modifier = {
		unit_offense_mult = 0.1
		unit_defense_mult = 0.1
		unit_morale_loss_mult = -0.1
		unit_morale_recovery_mult = 0.1
		character_battle_condition_logistics_secured_mult = 0.2
		character_battle_condition_broken_supply_line_mult = -0.2
	}
	ai_weight = {
		value = 1
	}
	group = supplies
}

		military_formation_mobilization_speed_mult = 0.1

			limit = {
				scope:military_formation = {
